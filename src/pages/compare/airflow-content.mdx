export function CheckCell({ value }) {
	if (value === true) {
		return <Check className="w-5 h-5 text-emerald-500 mx-auto" />;
	}
	if (value === false) {
		return <X className="w-5 h-5 text-gray-400 mx-auto" />;
	}
	if (value === 'Enterprise') {
		return (
			<div className="flex flex-col items-center gap-1">
				<Check className="w-5 h-5 text-emerald-500" />
				<EnterpriseBadge />
			</div>
		);
	}
	if (value === 'Vendor') {
		return (
			<div className="flex flex-col items-center gap-1">
				<Check className="w-5 h-5 text-emerald-500" />
				<VendorBadge />
			</div>
		);
	}
	return <span className="text-sm text-amber-600 dark:text-amber-400 font-medium">{value}</span>;
}
