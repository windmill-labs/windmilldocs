export function StackAnimation({ activeTab = 0, onSelect }) {
	const c = 240;
	const r = 180;

	/* Rotate orbit so the active card lands at 0° (right edge, facing the detail panel) */
	const activeItem = orbitItems.find((it) => it.tabIndex === activeTab);
	const orbitRotation = -activeItem.baseAngle;

	return (
		<div className="relative w-[480px] h-[480px]">
			{/* Center Windmill logo — stays fixed */}
			<div className="absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2 z-10">
				<motion.div
					animate={{ rotate: 360 }}

				>
					<img src="/img/windmill.svg" alt="Windmill" className="w-20 h-20" />
				</motion.div>
			</div>

			{/* Rotating orbit */}
			<motion.div
				className="absolute inset-0"
				animate={{ rotate: orbitRotation }}
				transition={{ duration: 0.6, ease: [0.4, 0, 0.2, 1] }}
			>
				{/* Orbit ring */}
				<div className="absolute inset-[60px] rounded-full border border-gray-200 dark:border-gray-700/50" />

				{/* Lines + particles */}
				<svg className="absolute inset-0 w-full h-full" style={{ pointerEvents: 'none', overflow: 'visible' }}>
					{orbitItems.map((item, i) => {
						const rad = (item.baseAngle * Math.PI) / 180;
						const rStart = 40;
						const isActive = item.tabIndex === activeTab;
						/* Active line extends to the right edge of the container (touches the detail panel border) */
						const rEnd = isActive ? 240 : r;
						return (
							<line
								key={`line-${i}`}
								x1={c + rStart * Math.cos(rad)}
								y1={c + rStart * Math.sin(rad)}
								x2={c + rEnd * Math.cos(rad)}
								y2={c + rEnd * Math.sin(rad)}
								className="stroke-gray-200 dark:stroke-gray-700/50"
								strokeWidth={1.5}
							/>
						);
					})}
					{(() => {
						const rDot = 220;
						const toXY = (angle) => {
							const rad = (angle * Math.PI) / 180;
							return { x: c + rDot * Math.cos(rad), y: c + rDot * Math.sin(rad) };
						};
						const pos = orbitItems.map((it) => toXY(it.baseAngle));
						const balls = [
							{ route: [0, 1, 2], duration: 10, delay: 0 },
							{ route: [2, 0, 1], duration: 12, delay: 2 },
							{ route: [1, 2, 0], duration: 11, delay: 4 },
							{ route: [0, 2, 1], duration: 13, delay: 6 },
							{ route: [2, 1, 0], duration: 10, delay: 8 },
							{ route: [1, 0, 2], duration: 12, delay: 10 },
						];
						return balls.map((b, i) => {
							const xK = [c];
							const yK = [c];
							b.route.forEach((idx) => {
								xK.push(pos[idx].x, c);
								yK.push(pos[idx].y, c);
							});
							return (
								<motion.circle
									key={`dot-${i}`}
									r={4}
									className="fill-blue-400 dark:fill-blue-300"
									style={{ opacity: 0.8 }}
									animate={{ cx: xK, cy: yK }}
									transition={{ duration: b.duration, repeat: Infinity, ease: 'linear', delay: b.delay }}
								/>
							);
						});
					})()}
				</svg>

				{/* Orbit cards — counter-rotate to stay upright, active card fades out at the right edge */}
				{orbitItems.map((item) => {
					const rad = (item.baseAngle * Math.PI) / 180;
					const x = c + r * Math.cos(rad);
					const y = c + r * Math.sin(rad);
					const isActive = item.tabIndex === activeTab;
					return (
						<div
							key={item.label}
							className="absolute"
							style={{ left: x, top: y, transform: 'translate(-50%, -50%)' }}
						>
							<motion.div
								className={isActive ? '' : 'cursor-pointer'}
								animate={{
									rotate: -orbitRotation,
									opacity: isActive ? 0 : 1,
									scale: isActive ? 0.8 : 1,
								}}
								transition={{ duration: 0.6, ease: [0.4, 0, 0.2, 1] }}
								onClick={() => !isActive && onSelect?.(item.tabIndex)}
							>

export function BoilerplateAnimation() {
	const [view, setView] = useState('local');
	const [openLayer, setOpenLayer] = useState(null);

	/* Animation phases: typing → connector → layers */
	const sectionRef = useRef(null);
	const isInView = useInView(sectionRef, { once: true, amount: 0.3 });
	const [phase, setPhase] = useState(0); /* 0=idle, 1=typing, 2=connector, 3=layers */
	const [typedChars, setTypedChars] = useState(0);

	/* Flatten all code lines into a single character count for the typing effect */
	const allText = codeLines.map((line) => line.tokens.map((t) => t.text).join('') || ' ').join('\n');
	const totalChars = allText.length;

	useEffect(() => {
		if (isInView && phase === 0) setPhase(1);
	}, [isInView]);

	/* Phase 1: typing effect */
	useEffect(() => {
		if (phase !== 1) return;
		const speed = 12; /* ms per character */
		const interval = setInterval(() => {
			setTypedChars((prev) => {
				if (prev >= totalChars) {
					clearInterval(interval);
					setTimeout(() => setPhase(2), 400);
					return totalChars;
				}
				return prev + 1;
			});
		}, speed);
		return () => clearInterval(interval);
	}, [phase, totalChars]);

	/* Phase 2 → 3: connector appears, then after a beat, layers cascade in */
	useEffect(() => {
		if (phase === 2) {
			const t = setTimeout(() => setPhase(3), 800);
			return () => clearTimeout(t);
		}
	}, [phase]);

	/* Helper: render code lines with typing effect */
	const renderTypedLines = () => {
		let charsSoFar = 0;
		return codeLines.map((line, i) => {
			const lineText = line.tokens.map((t) => t.text).join('') || ' ';
			const lineStart = charsSoFar;
			charsSoFar += lineText.length + 1; /* +1 for \n */
			const lineVisible = typedChars > lineStart;
			const lineCharsShown = Math.max(0, Math.min(lineText.length, typedChars - lineStart));
			if (!lineVisible) return (
				<div key={i} className="flex h-[1.75em]">
					<span className="text-gray-600 w-8 text-right mr-4 select-none opacity-0">{i + 1}</span>
				</div>
			);
			/* Render tokens up to lineCharsShown characters */
			let tokCharsSoFar = 0;
			const renderedTokens = [];
			for (const token of line.tokens) {
				if (tokCharsSoFar >= lineCharsShown) break;
				const visibleLen = Math.min(token.text.length, lineCharsShown - tokCharsSoFar);
				renderedTokens.push(
					<span key={renderedTokens.length} className={token.color} style={{ whiteSpace: 'pre' }}>{token.text.slice(0, visibleLen)}</span>
				);
				tokCharsSoFar += token.text.length;
			}
			return (
				<div key={i} className="flex">
					<span className="text-gray-600 w-8 text-right mr-4 select-none">{i + 1}</span>
					{line.tokens.length === 0 ? <span>&nbsp;</span> : renderedTokens}
					{lineCharsShown < lineText.length && (
						<span className="inline-block w-[2px] h-[1.1em] bg-gray-300 ml-[1px] animate-pulse" />
					)}
				</div>
			);
		});
	};

	return (
		<div ref={sectionRef} className="grid grid-cols-1 lg:grid-cols-2 gap-8 items-start">
			{/* Left: code editor + connector */}
			<div className="flex flex-col">
			<div className="rounded-2xl border border-gray-200 dark:border-gray-700 bg-gray-950 overflow-hidden flex-1 flex flex-col">
				<div className="flex items-center gap-2 px-4 py-3 border-b border-gray-800 bg-gray-900">
					<div className="flex gap-1.5">
						<span className="w-3 h-3 rounded-full bg-red-500/80" />
						<span className="w-3 h-3 rounded-full bg-yellow-500/80" />
