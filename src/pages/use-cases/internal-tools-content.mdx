export const orbitAngles = [-90, 150, 30];
export const orbitRadius = 170;
export const orbitCenter = 240;

export function HeroAnimation() {
	return (
		<div className="relative w-[480px] h-[480px]">
			{/* Orbit ring */}
			<div className="absolute inset-[70px] rounded-full border border-gray-200 dark:border-gray-700/50" />

			{/* Center Windmill logo */}
			<div className="absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2 z-10">
				<motion.div
					animate={{ rotate: 360 }}
					transition={{ duration: 45, repeat: Infinity, ease: 'linear' }}
				>
					<img src="/img/windmill.svg" alt="Windmill" className="w-20 h-20" />
				</motion.div>
			</div>

			{/* Rotating orbit */}
			<motion.div
				className="absolute inset-0"
				style={{ transformOrigin: `${orbitCenter}px ${orbitCenter}px` }}
				animate={{ rotate: 360 }}
				transition={{ duration: 45, repeat: Infinity, ease: 'linear' }}
			>
				{/* Lines from center to each card */}
				<svg className="absolute inset-0 w-full h-full" style={{ pointerEvents: 'none' }}>
					{orbitAngles.map((angle, i) => {
						const rad = (angle * Math.PI) / 180;
						const rStart = 40;
						return (
							<line
								key={i}
								x1={orbitCenter + rStart * Math.cos(rad)}
								y1={orbitCenter + rStart * Math.sin(rad)}
								x2={orbitCenter + orbitRadius * Math.cos(rad)}
								y2={orbitCenter + orbitRadius * Math.sin(rad)}
								className="stroke-gray-200 dark:stroke-gray-700/50"
								strokeWidth={1.5}
							/>
						);
					})}
					{/* Animated data particles */}
					{(() => {
						const c = orbitCenter;
						const r = orbitRadius;
						const toXY = (angle) => {
							const rad = (angle * Math.PI) / 180;
							return { x: c + r * Math.cos(rad), y: c + r * Math.sin(rad) };
						};
						const pos = orbitAngles.map(toXY);
						const balls = [
							{ route: [0, 1, 2, 0, 2, 1], duration: 14, delay: 0 },
							{ route: [1, 2, 0, 1, 0, 2], duration: 17, delay: 2 },
							{ route: [2, 0, 1, 2, 1, 0], duration: 12, delay: 4 },
							{ route: [0, 2, 1, 0, 1, 2], duration: 16, delay: 6 },
							{ route: [1, 0, 2, 1, 2, 0], duration: 13, delay: 8 },
							{ route: [2, 1, 0, 2, 0, 1], duration: 18, delay: 10 },
						];
						return balls.map((b, i) => {
							const xK = [c];
							const yK = [c];
							b.route.forEach((idx) => {
								xK.push(pos[idx].x, c);
								yK.push(pos[idx].y, c);
							});
							return (
								<motion.circle
									key={`dot-${i}`}
									r={4}
									className="fill-blue-400 dark:fill-blue-300"
									style={{ opacity: 0.8 }}
									animate={{ cx: xK, cy: yK }}
									transition={{ duration: b.duration, repeat: Infinity, ease: 'linear', delay: b.delay }}
								/>
							);
						});
					})()}
				</svg>

				{/* Layer cards on the orbit */}
				{layerItems.map((layer, i) => {
					const rad = (orbitAngles[i] * Math.PI) / 180;
					const x = orbitCenter + orbitRadius * Math.cos(rad);
					const y = orbitCenter + orbitRadius * Math.sin(rad);
					return (
						<div
							key={layer.label}
							className="absolute"
							style={{ left: x, top: y, transform: 'translate(-50%, -50%)' }}
						>
							<motion.div
								animate={{ rotate: -360 }}
								transition={{ duration: 45, repeat: Infinity, ease: 'linear' }}
							>
								<div className={`relative rounded-2xl bg-gradient-to-r ${layer.color} p-[2px] shadow-lg`} style={{ width: 170 }}>
									<div className="rounded-[14px] bg-white dark:bg-gray-900 px-4 py-4 text-center">
										<div className={`w-10 h-10 rounded-xl bg-gradient-to-br ${layer.color} flex items-center justify-center mx-auto mb-2`}>
											{i === 0 && <LayoutGrid className="w-5 h-5 text-white" />}
											{i === 1 && <Database className="w-5 h-5 text-white" />}
											{i === 2 && <Code2 className="w-5 h-5 text-white" />}
										</div>
										<span className="text-base font-semibold text-gray-900 dark:text-white block">{layer.label}</span>
										<span className="text-xs text-gray-500 dark:text-gray-400 mt-0.5 block">{layer.sublabel}</span>
									</div>
								</div>
							</motion.div>
						</div>
					);
				})}
			</motion.div>
		</div>
	);
}
