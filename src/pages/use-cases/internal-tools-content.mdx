export const orbitItems = [
	{ Icon: LayoutGrid, label: 'Frontend', sublabel: 'React / Vue / Svelte', color: 'from-emerald-400 to-emerald-600', baseAngle: -90, tabIndex: 2 },

export function StackAnimation({ activeTab = 0, onSelect }) {
	const c = 240;
	const r = 180;

	/* Rotate orbit so the active card lands at 0° (right edge, facing the detail panel) */
	const activeItem = orbitItems.find((it) => it.tabIndex === activeTab);
	const orbitRotation = -activeItem.baseAngle;

	return (
		<div className="relative w-[480px] h-[480px]">
			{/* Center Windmill logo — stays fixed */}
			<div className="absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2 z-10">
				<motion.div
					animate={{ rotate: 360 }}
					transition={{ duration: 45, repeat: Infinity, ease: 'linear' }}
				>
					<img src="/img/windmill.svg" alt="Windmill" className="w-20 h-20" />
				</motion.div>
			</div>

			{/* Rotating orbit */}
			<motion.div
				className="absolute inset-0"
				animate={{ rotate: orbitRotation }}
				transition={{ duration: 0.6, ease: [0.4, 0, 0.2, 1] }}
			>
				{/* Orbit ring */}
				<div className="absolute inset-[60px] rounded-full border border-gray-200 dark:border-gray-700/50" />

				{/* Lines + particles */}
				<svg className="absolute inset-0 w-full h-full" style={{ pointerEvents: 'none', overflow: 'visible' }}>
					{orbitItems.map((item, i) => {
						const rad = (item.baseAngle * Math.PI) / 180;
						const rStart = 40;
						const isActive = item.tabIndex === activeTab;
						/* Active line extends to the right edge of the container (touches the detail panel border) */
						const rEnd = isActive ? 240 : r;
						return (
							<line
								key={`line-${i}`}
								x1={c + rStart * Math.cos(rad)}
								y1={c + rStart * Math.sin(rad)}
								x2={c + rEnd * Math.cos(rad)}
								y2={c + rEnd * Math.sin(rad)}
								className="stroke-gray-200 dark:stroke-gray-700/50"
								strokeWidth={1.5}
							/>
						);
					})}
					{(() => {
						const toXY = (angle) => {
							const rad = (angle * Math.PI) / 180;
							return { x: c + r * Math.cos(rad), y: c + r * Math.sin(rad) };
						};
						const pos = orbitItems.map((it) => toXY(it.baseAngle));
						const balls = [
							{ route: [0, 1, 2], duration: 10, delay: 0 },
							{ route: [2, 0, 1], duration: 12, delay: 2 },
							{ route: [1, 2, 0], duration: 11, delay: 4 },
							{ route: [0, 2, 1], duration: 13, delay: 6 },
							{ route: [2, 1, 0], duration: 10, delay: 8 },
							{ route: [1, 0, 2], duration: 12, delay: 10 },
						];
						return balls.map((b, i) => {
							const xK = [c];
							const yK = [c];
							b.route.forEach((idx) => {
								xK.push(pos[idx].x, c);
								yK.push(pos[idx].y, c);
							});
							return (
								<motion.circle
									key={`dot-${i}`}
									r={4}
									className="fill-blue-400 dark:fill-blue-300"
									style={{ opacity: 0.8 }}
									animate={{ cx: xK, cy: yK }}
									transition={{ duration: b.duration, repeat: Infinity, ease: 'linear', delay: b.delay }}
								/>
							);
						});
					})()}
				</svg>

				{/* Orbit cards — counter-rotate to stay upright, active card fades out at the right edge */}
				{orbitItems.map((item) => {
					const rad = (item.baseAngle * Math.PI) / 180;
					const x = c + r * Math.cos(rad);
					const y = c + r * Math.sin(rad);
					const isActive = item.tabIndex === activeTab;
					return (
						<div
							key={item.label}
							className="absolute"
							style={{ left: x, top: y, transform: 'translate(-50%, -50%)' }}
						>
							<motion.div
								className={isActive ? '' : 'cursor-pointer'}
								animate={{
									rotate: -orbitRotation,
									opacity: isActive ? 0 : 1,
									scale: isActive ? 0.8 : 1,
								}}
								transition={{ duration: 0.6, ease: [0.4, 0, 0.2, 1] }}
								onClick={() => !isActive && onSelect?.(item.tabIndex)}
							>
								<div className={`relative rounded-2xl bg-gradient-to-r ${item.color} p-[2px] shadow-lg`}>
									<div className="rounded-[14px] bg-white dark:bg-gray-900 px-5 py-4 flex items-center gap-4 whitespace-nowrap">
										<div className={`w-11 h-11 rounded-lg bg-gradient-to-br ${item.color} flex items-center justify-center flex-shrink-0`}>
											<item.Icon className="w-5 h-5 text-white" />
										</div>
										<div>
											<span className="text-base font-semibold text-gray-900 dark:text-white block">{item.label}</span>
											<span className="text-xs text-gray-500 dark:text-gray-400 block">{item.sublabel}</span>
										</div>
									</div>
								</div>
							</motion.div>
						</div>
					);
				})}
			</motion.div>
		</div>
	);
}
