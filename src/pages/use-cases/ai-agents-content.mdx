export const heroNodes = heroNodeDefs.map((node, i) => {
	const angle = (-90 + i * 45) * (Math.PI / 180);
	return { ...node, x: Math.round(center.x + radius * Math.cos(angle)), y: Math.round(center.y + radius * Math.sin(angle)) };
});

{/* All-to-all connections between outer nodes for neural-network web */}
export const crossLinks = heroNodes.flatMap((_, i) =>
	heroNodes.slice(i + 1).map((_, j) => [i, i + 1 + j])
);

export function HeroAnimation() {
	return (
		<div className="relative w-[480px] h-[480px]">
			{/* Center Bot icon */}
			<div className="absolute z-10" style={{ left: center.x, top: center.y, transform: 'translate(-50%, -50%)' }}>
				<div className="w-20 h-20 rounded-2xl bg-gradient-to-br from-blue-500 to-blue-900 flex items-center justify-center shadow-lg">
					<Bot className="w-10 h-10 text-white" />
				</div>
			</div>

			{/* SVG connections + particles */}
			<svg className="absolute inset-0 w-full h-full" style={{ pointerEvents: 'none' }}>
				{/* Web connections between all outer nodes */}
				{crossLinks.map(([a, b], i) => (
					<line
						key={`x-${i}`}
						x1={heroNodes[a].x} y1={heroNodes[a].y}
						x2={heroNodes[b].x} y2={heroNodes[b].y}
						className="stroke-gray-300/40 dark:stroke-gray-600/25"
						strokeWidth={0.75}
					/>
				))}
				{/* Lines from each node to center */}
				{heroNodes.map((node, i) => (
					<line
						key={`c-${i}`}
						x1={center.x} y1={center.y}
						x2={node.x} y2={node.y}
						className="stroke-gray-300 dark:stroke-gray-600/60"
						strokeWidth={1.5}
					/>
				))}
				{/* Animated particles traveling along the web */}
				{heroNodes.map((node, i) => {
					const xK = [center.x, node.x, center.x, heroNodes[(i + 3) % heroNodes.length].x, center.x];
					const yK = [center.y, node.y, center.y, heroNodes[(i + 3) % heroNodes.length].y, center.y];
					return (
						<motion.circle
							key={`dot-${i}`}
							r={3.5}
							className="fill-blue-400 dark:fill-blue-300"
							style={{ opacity: 0.7 }}
							animate={{ cx: xK, cy: yK }}
							transition={{ duration: 10 + i * 1.5, repeat: Infinity, ease: 'linear', delay: i * 1.2 }}
						/>
					);
				})}
			</svg>

			{/* Service nodes */}
			{heroNodes.map((node) => (
				<div
					key={node.label}
					className="absolute z-10"
					style={{ left: node.x, top: node.y, transform: 'translate(-50%, -50%)' }}
				>
					<div className="flex flex-col items-center gap-1">
						<div className="w-14 h-14 rounded-xl bg-white border border-gray-200 dark:border-gray-600 shadow-sm flex items-center justify-center p-2">
							<img src={node.src} alt={node.label} className="w-full h-full object-contain" />
						</div>
						<span className="text-[10px] font-medium text-gray-500 dark:text-gray-400">{node.label}</span>
					</div>
				</div>
			))}
		</div>
	);
}
