export const dagNodeMap = Object.fromEntries(dagNodes.map(n => [n.id, n]));

export function edgePath(e) {
	const s = dagNodeMap[e.src];
	const d = dagNodeMap[e.dst];
	const sy = s.y + NH / 2;
	const dy = d.y - NH / 2;
	if (s.x === d.x) return `M${s.x},${sy} L${d.x},${dy}`;
	const midY = (sy + dy) / 2;
	return `M${s.x},${sy} C${s.x},${midY} ${d.x},${midY} ${d.x},${dy}`;
}
