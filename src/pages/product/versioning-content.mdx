export function SyncFlows() {
	const ref = useRef(null);
	const isInView = useInView(ref, { once: true, margin: "-40px" });
	const d = 0.2;
	const flows = [
		{
			label: 'From local to production',
			steps: [
				{ icon: Terminal, label: 'wmill sync push', auto: false },
			],
			parallel: [
				{ icon: Rocket, label: 'Deploy on Windmill', auto: true },
				{ icon: GitBranch, label: 'Commit to Git', auto: true },
			],
			after: [
				{ icon: Monitor, label: 'Available on the UI', auto: true },
			],
		},
		{
			label: 'From UI to local',
			steps: [
				{ icon: Monitor, label: 'Deploy from the UI', auto: false },
				{ icon: GitBranch, label: 'Commit to Git', auto: true },
				{ icon: Terminal, label: 'wmill sync pull', auto: false },
				{ icon: PenLine, label: 'Files updated locally', auto: true },
			],
		},
	];
	return (
		<div ref={ref} className="mt-8 grid grid-cols-1 md:grid-cols-2 gap-6">
			{flows.map((flow, fi) => {
				const baseDelay = d + fi * 0.4;
				return (
					<motion.div
						key={flow.label}
						initial={{ opacity: 0, y: 12 }}
						animate={isInView ? { opacity: 1, y: 0 } : {}}
						transition={{ duration: 0.35, delay: baseDelay }}
						className="p-5 rounded-xl border border-gray-200 dark:border-gray-700 bg-white/5 dark:bg-gray-900/50"
					>
						<p className="text-xs font-medium text-gray-400 dark:text-gray-500 uppercase tracking-wider mb-3">
							{flow.label}
						</p>
						<div className="flex flex-col gap-1.5">
							{flow.steps.map((step, i) => (
								<React.Fragment key={step.label}>
									<SyncFlowStep {...step} isInView={isInView} delay={baseDelay + 0.1 + i * 0.12} />
									{(i < flow.steps.length - 1 || flow.parallel) && (
										<div className="flex justify-center">
											<svg width="12" height="12" viewBox="0 0 12 12" className="text-gray-400 dark:text-gray-500">
												<path d="M6 2 L6 10 M3 7 L6 10 L9 7" fill="none" stroke="currentColor" strokeWidth="1.5" strokeLinecap="round" strokeLinejoin="round" />
											</svg>
										</div>
									)}
								</React.Fragment>
							))}
							{flow.parallel && (
								<div className="relative">
									{/* Split curves */}
									<svg className="w-full" height="32" viewBox="0 0 200 32" preserveAspectRatio="xMidYMid meet">
										<path d="M100,0 L100,4 C100,18 50,18 50,32" fill="none" stroke="currentColor" strokeWidth="1" className="text-gray-300 dark:text-gray-600" />
										<path d="M100,0 L100,4 C100,18 150,18 150,32" fill="none" stroke="currentColor" strokeWidth="1" className="text-gray-300 dark:text-gray-600" />
									</svg>
									<div className="flex gap-2">
										{flow.parallel.map((step, i) => (
											<div key={step.label} className="flex-1">
												<SyncFlowStep {...step} isInView={isInView} delay={baseDelay + 0.1 + flow.steps.length * 0.12 + i * 0.08} />
											</div>
										))}
									</div>
									{flow.after && (
										<>
											{/* Merge curves */}
											<svg className="w-full" height="32" viewBox="0 0 200 32" preserveAspectRatio="xMidYMid meet">
												<path d="M50,0 C50,14 100,14 100,28 L100,32" fill="none" stroke="currentColor" strokeWidth="1" className="text-gray-300 dark:text-gray-600" />
												<path d="M150,0 C150,14 100,14 100,28 L100,32" fill="none" stroke="currentColor" strokeWidth="1" className="text-gray-300 dark:text-gray-600" />
											</svg>
											{/* Arrow pointing down */}
											<div className="flex justify-center -mt-1">
												<svg width="12" height="12" viewBox="0 0 12 12" className="text-gray-400 dark:text-gray-500">
													<path d="M6 2 L6 10 M3 7 L6 10 L9 7" fill="none" stroke="currentColor" strokeWidth="1.5" strokeLinecap="round" strokeLinejoin="round" />
												</svg>
											</div>
										</>
									)}
								</div>
							)}
							{flow.after && flow.after.map((step, i) => (
								<SyncFlowStep key={step.label} {...step} isInView={isInView} delay={baseDelay + 0.1 + (flow.steps.length + flow.parallel.length) * 0.12 + i * 0.08} />
							))}
						</div>
					</motion.div>
				);
			})}
		</div>
	);
}
