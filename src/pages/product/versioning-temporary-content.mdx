{/* ── Hero animation: Edit → Deploy → Versioned → Synced to Git ── */}

export function HeroTimeline() {
	const ref = useRef(null);
	const isInView = useInView(ref, { once: true, margin: "-40px" });
	const d = 0.3;
	const steps = [
		{ icon: PenLine, label: 'Edit in UI or locally' },
		{ icon: Rocket, label: 'Deploy' },
		{ icon: History, label: 'Versioned' },
		{ icon: GitBranch, label: 'Synced to Git' },
	];
	return (
		<div ref={ref} className="mt-12 mb-4">
			{/* Desktop */}
			<div className="hidden sm:flex items-center justify-center gap-0">
				{steps.map((step, i) => (
					<React.Fragment key={step.label}>
						<motion.div
							initial={{ opacity: 0, y: 12 }}
							animate={isInView ? { opacity: 1, y: 0 } : {}}
							transition={{ duration: 0.3, delay: d + i * 0.15 }}
							className="flex items-center gap-2.5 px-4 py-2.5 rounded-lg border border-gray-200 dark:border-gray-700 bg-white/5 dark:bg-gray-900/50"
						>
							<step.icon className="w-4 h-4 text-gray-500 dark:text-gray-400 flex-shrink-0" />
							<span className="text-sm font-medium text-gray-700 dark:text-gray-300 whitespace-nowrap">{step.label}</span>
						</motion.div>
						{i < steps.length - 1 && (
							<motion.div
								initial={{ opacity: 0, scaleX: 0 }}
								animate={isInView ? { opacity: 1, scaleX: 1 } : {}}
								transition={{ duration: 0.15, delay: d + i * 0.15 + 0.1 }}
								className="mx-1 origin-left"
							>
								<ChevronRight className="w-4 h-4 text-gray-400 dark:text-gray-500" />
							</motion.div>
						)}
					</React.Fragment>
				))}
			</div>
			{/* Mobile */}
			<div className="sm:hidden flex flex-wrap justify-center gap-2">
				{steps.map((step, i) => (
					<motion.div
						key={step.label}
						initial={{ opacity: 0, y: 8 }}
						animate={isInView ? { opacity: 1, y: 0 } : {}}
						transition={{ duration: 0.25, delay: d + i * 0.1 }}
						className="flex items-center gap-2 px-3 py-2 rounded-lg border border-gray-200 dark:border-gray-700 bg-white/5 dark:bg-gray-900/50"
					>
						<step.icon className="w-3.5 h-3.5 text-gray-500 dark:text-gray-400 flex-shrink-0" />
						<span className="text-xs font-medium text-gray-700 dark:text-gray-300">{step.label}</span>
					</motion.div>
				))}
			</div>
		</div>
	);
}

	return (
		<motion.div
			initial={{ opacity: 0, scaleX: 0 }}

export function SyncFlows() {
	const ref = useRef(null);
	const isInView = useInView(ref, { once: true, margin: "-40px" });
	const d = 0.2;
	const flows = [
		{
			label: 'From local to production',
			steps: [
				{ icon: Terminal, label: 'wmill sync push' },
			],
			parallel: [
				{ icon: Rocket, label: 'Deploy on Windmill' },
				{ icon: GitBranch, label: 'Commit to Git' },
			],
			after: [
				{ icon: Monitor, label: 'New version on the UI' },
			],
		},
		{
			label: 'From UI to local',
			steps: [
				{ icon: Monitor, label: 'Deploy from the UI' },
				{ icon: GitBranch, label: 'Commit to Git' },
				{ icon: Terminal, label: 'wmill sync pull' },
				{ icon: PenLine, label: 'Update locally' },
			],
		},
	];
	return (
		<div ref={ref} className="mt-8 grid grid-cols-1 md:grid-cols-2 gap-6">
			{flows.map((flow, fi) => {
				const baseDelay = d + fi * 0.4;
				return (
					<motion.div
						key={flow.label}
						initial={{ opacity: 0, y: 12 }}
						animate={isInView ? { opacity: 1, y: 0 } : {}}
						transition={{ duration: 0.35, delay: baseDelay }}
						className="p-5 rounded-xl border border-gray-200 dark:border-gray-700 bg-white/5 dark:bg-gray-900/50"
					>
						<p className="text-xs font-medium text-gray-400 dark:text-gray-500 uppercase tracking-wider mb-3">
							{flow.label}
						</p>
						<div className="flex flex-col gap-1.5">
							{flow.steps.map((step, i) => (
								<React.Fragment key={step.label}>
									<SyncFlowStep {...step} isInView={isInView} delay={baseDelay + 0.1 + i * 0.12} />
									{(i < flow.steps.length - 1 || flow.parallel) && (
										<div className="flex justify-center">
											<svg width="12" height="12" viewBox="0 0 12 12" className="text-gray-400 dark:text-gray-500">
												<path d="M6 2 L6 10 M3 7 L6 10 L9 7" fill="none" stroke="currentColor" strokeWidth="1.5" strokeLinecap="round" strokeLinejoin="round" />
											</svg>
										</div>
									)}
								</React.Fragment>
							))}
							{flow.parallel && (
								<div className="relative">
									{/* Split curves */}
									<svg className="w-full" height="32" viewBox="0 0 200 32" preserveAspectRatio="xMidYMid meet">
										<path d="M100,0 L100,4 C100,18 50,18 50,32" fill="none" stroke="currentColor" strokeWidth="1" className="text-gray-300 dark:text-gray-600" />
										<path d="M100,0 L100,4 C100,18 150,18 150,32" fill="none" stroke="currentColor" strokeWidth="1" className="text-gray-300 dark:text-gray-600" />
									</svg>
									<div className="flex gap-2">
										{flow.parallel.map((step, i) => (
											<div key={step.label} className="flex-1">
												<SyncFlowStep {...step} isInView={isInView} delay={baseDelay + 0.1 + flow.steps.length * 0.12 + i * 0.08} />
											</div>
										))}
									</div>
									{flow.after && (
										<>
											{/* Merge curves */}
											<svg className="w-full" height="32" viewBox="0 0 200 32" preserveAspectRatio="xMidYMid meet">
												<path d="M50,0 C50,14 100,14 100,28 L100,32" fill="none" stroke="currentColor" strokeWidth="1" className="text-gray-300 dark:text-gray-600" />
												<path d="M150,0 C150,14 100,14 100,28 L100,32" fill="none" stroke="currentColor" strokeWidth="1" className="text-gray-300 dark:text-gray-600" />
											</svg>
											{/* Arrow pointing down */}
											<div className="flex justify-center -mt-1">
												<svg width="12" height="12" viewBox="0 0 12 12" className="text-gray-400 dark:text-gray-500">
													<path d="M6 2 L6 10 M3 7 L6 10 L9 7" fill="none" stroke="currentColor" strokeWidth="1.5" strokeLinecap="round" strokeLinejoin="round" />
												</svg>
											</div>
										</>
									)}
								</div>
							)}
							{flow.after && flow.after.map((step, i) => (
								<SyncFlowStep key={step.label} {...step} isInView={isInView} delay={baseDelay + 0.1 + (flow.steps.length + flow.parallel.length) * 0.12 + i * 0.08} />
							))}
						</div>
					</motion.div>
				);
			})}
		</div>
	);
}
