export function HeroAnimation() {
	const ref = useRef(null);
	const isInView = useInView(ref, { once: true, amount: 0.2 });
	const [visibleDots, setVisibleDots] = useState(0);
	const [visibleRows, setVisibleRows] = useState(0);

	useEffect(() => {
		if (!isInView) return;
		const dotInterval = setInterval(() => {
			setVisibleDots((prev) => {
				if (prev >= scatterPoints.length) { clearInterval(dotInterval); return scatterPoints.length; }
				return prev + 1;
			});
		}, 50);
		const rowTimeout = setTimeout(() => {
			const rowInterval = setInterval(() => {
				setVisibleRows((prev) => {
					if (prev >= jobRuns.length) { clearInterval(rowInterval); return jobRuns.length; }
					return prev + 1;
				});
			}, 250);
		}, 600);
		return () => { clearInterval(dotInterval); clearTimeout(rowTimeout); };
	}, [isInView]);

	return (
		<div ref={ref} className="rounded-2xl border border-gray-700 bg-gray-950 overflow-hidden shadow-lg">
			{/* Toolbar */}
			<div className="flex items-center gap-2 px-4 py-2 border-b border-gray-800">
				<span className="text-sm font-bold text-white">Runs</span>
				<span className="px-2 py-0.5 rounded bg-blue-600/20 text-blue-400 text-[10px] font-medium">All</span>
				<span className="text-[10px] text-gray-500">Runs</span>
				<span className="text-[10px] text-gray-500">Deps</span>
				<div className="flex-1" />
				<CheckCircle2 className="w-3.5 h-3.5 text-emerald-400" />
				<XCircle className="w-3.5 h-3.5 text-red-400/60" />
				<Clock className="w-3.5 h-3.5 text-gray-600" />
			</div>

			{/* Scatter chart */}
			<div className="px-4 pt-2 pb-2 border-b border-gray-800">
				<div className="flex items-center gap-3 mb-2">
					<span className="text-[11px] font-medium text-white">Duration</span>
					<span className="text-[11px] text-gray-500">Concurrency</span>
				</div>
				<div className="relative w-full" style={{ height: chartH }}>
					{yTicks.map((tick) => (
						<div key={tick} className="absolute left-0 right-0 flex items-center" style={{ top: yPos(tick) }}>
							<span className="text-[10px] text-gray-500 tabular-nums w-10 text-right shrink-0">{tick >= 1000 ? `${tick/1000} 000` : tick}</span>
							<div className="ml-2 h-px bg-gray-800 flex-1" />
						</div>
					))}
					<svg className="absolute top-0 bottom-0" style={{ left: 48, right: 0, width: 'calc(100% - 48px)', height: chartH }}>
						{scatterPoints.map((pt, i) => (
							<motion.circle
								key={i}
								cx={`${pt.x * 100}%`}
								cy={yPos(pt.y)}
								className={pt.ok ? 'fill-emerald-400' : 'fill-red-400'}
								initial={{ opacity: 0, r: 0 }}
								animate={i < visibleDots ? { opacity: 0.85, r: 4 } : { opacity: 0, r: 0 }}
								transition={{ duration: 0.15 }}
							/>
						))}
					</svg>
				</div>
			</div>

			{/* Table header */}
			<div className="grid px-4 py-1.5 border-b border-gray-800 text-[10px] text-gray-500 font-medium" style={{ gridTemplateColumns: '20px 1fr 48px 1fr 32px' }}>
				<span></span>
				<span>25+ jobs</span>
				<span>Duration</span>
				<span>Triggered by</span>
				<span>Tag</span>
			</div>

			{/* Date header */}
			<motion.div
				className="px-4 py-1 bg-gray-900 border-b border-gray-800 text-[10px] font-semibold text-gray-300"
				initial={{ opacity: 0 }}
				animate={visibleRows > 0 ? { opacity: 1 } : { opacity: 0 }}
				transition={{ duration: 0.3 }}
			>
				March 12, 2026
			</motion.div>

			{/* Rows */}
			{jobRuns.map((job, i) => (
				<motion.div
					key={job.id}
					className="grid px-4 py-1.5 border-b border-gray-800/50 items-center"
					style={{ gridTemplateColumns: '20px 1fr 48px 1fr 32px' }}
					initial={{ opacity: 0, y: 5 }}
					animate={i < visibleRows ? { opacity: 1, y: 0 } : { opacity: 0, y: 5 }}
					transition={{ duration: 0.2 }}
				>
					{job.ok
						? <CheckCircle2 className="w-3.5 h-3.5 text-emerald-400" />
						: <XCircle className="w-3.5 h-3.5 text-red-400" />
					}
					<span className="text-[10px] text-gray-400 truncate">{job.ended}</span>
					<span className="text-[10px] text-gray-300 tabular-nums">{job.duration}</span>
					<div className="flex items-center gap-1 truncate">
						<Calendar className="w-2.5 h-2.5 text-gray-600 shrink-0" />
						<span className="text-[10px] text-gray-400 truncate font-mono">{job.triggeredBy}</span>
					</div>
					<span className="text-[10px] text-gray-500">{job.tag}</span>
				</motion.div>
			))}
		</div>
	);
}
