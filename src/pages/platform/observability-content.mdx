export function HeroAnimation() {
	const ref = useRef(null);
	const isInView = useInView(ref, { once: true, amount: 0.2 });
	const [visibleDots, setVisibleDots] = useState(0);
	const [visibleRows, setVisibleRows] = useState(0);

	useEffect(() => {
		if (!isInView) return;
		const dotInterval = setInterval(() => {
			setVisibleDots((prev) => {
				if (prev >= scatterPoints.length) { clearInterval(dotInterval); return scatterPoints.length; }
				return prev + 1;
			});
		}, 50);
		const rowTimeout = setTimeout(() => {
			const rowInterval = setInterval(() => {
				setVisibleRows((prev) => {
					if (prev >= jobRuns.length) { clearInterval(rowInterval); return jobRuns.length; }
					return prev + 1;
				});
			}, 250);
		}, 600);
		return () => { clearInterval(dotInterval); clearTimeout(rowTimeout); };
	}, [isInView]);

	return (
		<div ref={ref} className="rounded-2xl border border-gray-200 bg-white overflow-hidden shadow-lg">
			{/* Toolbar */}
			<div className="flex items-center gap-2 px-4 py-2 border-b border-gray-200">
				<span className="text-sm font-bold text-gray-900">Runs</span>
				<span className="px-2 py-0.5 rounded bg-blue-600/20 text-blue-600 text-[10px] font-medium">All</span>
				<span className="text-[10px] text-gray-400">Runs</span>
				<span className="text-[10px] text-gray-400">Deps</span>
				<div className="flex-1" />
				<CheckCircle2 className="w-3.5 h-3.5 text-emerald-500" />
				<XCircle className="w-3.5 h-3.5 text-red-400/60" />
				<Clock className="w-3.5 h-3.5 text-gray-400" />
			</div>

			{/* Scatter chart */}
			<div className="px-4 pt-2 pb-2">
				<div className="flex items-center gap-3 mb-2">
					<span className="text-[11px] font-medium text-gray-900">Duration</span>
					<span className="text-[11px] text-gray-400">Concurrency</span>
				</div>
				<div className="relative w-full" style={{ height: chartH }}>
					{yTicks.map((tick) => (
						<div key={tick} className="absolute left-0 right-0 flex items-center" style={{ top: yPos(tick) }}>
							<span className="text-[10px] text-gray-400 tabular-nums w-10 text-right shrink-0">{tick >= 1000 ? `${tick/1000} 000` : tick}</span>
							<div className="ml-2 h-px bg-gray-200 flex-1" />
						</div>
					))}
					<svg className="absolute top-0 bottom-0" style={{ left: 48, right: 0, width: 'calc(100% - 48px)', height: chartH }}>
						{scatterPoints.map((pt, i) => (
							<motion.circle
								key={i}
								cx={`${pt.x * 100}%`}
								cy={yPos(pt.y)}
								className={pt.ok ? 'fill-emerald-500' : 'fill-red-500'}
								initial={{ opacity: 0, r: 0 }}
								animate={i < visibleDots ? { opacity: 0.85, r: 4 } : { opacity: 0, r: 0 }}
								transition={{ duration: 0.15 }}
							/>
						))}
					</svg>
				</div>
			</div>

			{/* Table header */}
			<motion.div
				className="grid px-4 py-1 mt-2 bg-gray-100 border-b border-gray-200 text-[10px] font-medium text-gray-500"
				style={{ gridTemplateColumns: '20px 1fr 48px 1fr 32px' }}
				initial={{ opacity: 0 }}
				animate={visibleRows > 0 ? { opacity: 1 } : { opacity: 0 }}
				transition={{ duration: 0.3 }}
			>
				<span></span>
				<span>Path</span>
				<span>Duration</span>
				<span>Triggered by</span>
				<span>Tag</span>
			</motion.div>

			{/* Rows */}
			{jobRuns.map((job, i) => (
				<motion.div
					key={job.id}
					className="grid px-4 py-1.5 border-b border-gray-100 items-center"
					style={{ gridTemplateColumns: '20px 1fr 48px 1fr 32px' }}
					initial={{ opacity: 0, y: 5 }}
					animate={i < visibleRows ? { opacity: 1, y: 0 } : { opacity: 0, y: 5 }}
					transition={{ duration: 0.2 }}
				>
					{job.ok
						? <CheckCircle2 className="w-3.5 h-3.5 text-emerald-500" />
						: <XCircle className="w-3.5 h-3.5 text-red-500" />
					}
					<span className="text-[10px] text-gray-700 truncate font-mono">{job.name}</span>
					<span className="text-[10px] text-gray-700 tabular-nums">{job.duration}</span>
					<div className="flex items-center gap-1 truncate">
						{job.triggeredBy === 'Schedule' && <Calendar className="w-2.5 h-2.5 text-gray-400 shrink-0" />}
						{job.triggeredBy === 'Webhook' && <Webhook className="w-2.5 h-2.5 text-gray-400 shrink-0" />}
						{job.triggeredBy === 'Kafka' && <Radio className="w-2.5 h-2.5 text-gray-400 shrink-0" />}
						<span className="text-[10px] text-gray-500 truncate font-mono">{job.triggeredBy}</span>
					</div>
					<span className="text-[10px] text-gray-400">{job.tag}</span>
				</motion.div>
			))}
		</div>
	);
}
